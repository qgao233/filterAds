function traverseAndRemove(node) {    
    if (node.nodeType === Node.ELEMENT_NODE && node.tagName) {    
        if (node.tagName.toLowerCase() === 'cs-native-ad-card') {    
            node.parentNode.removeChild(node);    
            return;    
        }    
        if (node.shadowRoot) {    
            traverseShadowDOM(node.shadowRoot);    
        }    
    }    
    if (node.childNodes) {    
        for (let childNode of node.childNodes) {    
            traverseAndRemove(childNode);    
        }    
    }    
}    
    
function traverseShadowDOM(shadowRoot) {    
    for (let node of shadowRoot.childNodes) {    
        traverseAndRemove(node);    
    }    
}
